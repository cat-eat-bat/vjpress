  settings:
    encrypt: 
        enabled: true  # 开启连接参数加密
        scopes: password #  指定 jdbc 参数加密范围，支持： url, username，用户名加密 password（默认）密码加密

    show-sql: true  # 在日志里输出 sql 语句
    entity-packages: cn.piesat.v.sample.sorm.domain.entity
    dev-mode: true  # 开发模式 生产环境不建议开启
